ğŸ¯ Smart Pointers (C++11)
#include <memory>
// unique_ptr - exclusive ownership
unique_ptr<int> ptr1 = make_unique<int>(42);
// Can't copy, only move
unique_ptr<int> ptr2 = move(ptr1);
// shared_ptr - shared ownership
shared_ptr<int> sp1 = make_shared<int>(42);
shared_ptr<int> sp2 = sp1; // Both own the object
// weak_ptr - non-owning reference
weak_ptr<int> wp = sp1;
ğŸ”§ References
// Reference - alias to existing variable
int x = 10;
int &ref = x; // ref is alias for x
ref = 20; // x is now 20
// Function with reference parameter
void increment(int &n) {
n++; // Modifies original
}
// Const reference (efficient for large objects)
void print(const string &str) {
cout << str; // Can't modify str
}

private:

public:

elements.push_back(elem);
}
T pop() {
T elem = elements.back();
elements.pop_back();
return elem;
}
};
