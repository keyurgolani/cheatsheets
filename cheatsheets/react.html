ü™ù useState Hook
const [state, setState] = useState(initialValue)
Declare state variable with initial value
// Basic usage
const [count, setCount] = useState(0);
setCount(count + 1);
// Functional update (when new state depends on previous)
setCount(prevCount => prevCount + 1);
// Lazy initialization (expensive computation)
const [state, setState] = useState(() => {
return expensiveComputation();
});
üí° Use functional updates when new state depends on previous state to avoid stale closures!
‚ö° useEffect Hook
useEffect(() => { /* effect */ }, [deps])
Run side effects after render
// Run on every render
useEffect(() => {
console.log('Runs after every render');
});
// Run only once (on mount)
useEffect(() => {
console.log('Runs once on mount');
}, []);
// Run when dependencies change
useEffect(() => {
console.log('Count changed:', count);
}, [count]);
// Cleanup function
useEffect(() => {
const timer = setInterval(() => {}, 1000);
return () => clearInterval(timer);
}, []);
‚ö†Ô∏è Always include all dependencies in the dependency array to avoid bugs!
üéØ useContext Hook
const value = useContext(MyContext)
Access context value without Consumer
// Create context
const ThemeContext = React.createContext('light');
// Provider component
function App() {
return (
<ThemeContext.Provider value="dark">
<Toolbar />
</ThemeContext.Provider>
);
}
// Consumer component
function ThemedButton() {
const theme = useContext(ThemeContext);
return <button className={theme}>Click</button>;
}
üîÑ useReducer Hook
const [state, dispatch] = useReducer(reducer, initialState)
Manage complex state logic with reducer pattern
// Reducer function
function reducer(state, action) {
switch (action.type) {
case 'increment':
return { count: state.count + 1 };
case 'decrement':
return { count: state.count - 1 };
default:
return state;
}
}
// Usage
const [state, dispatch] = useReducer(reducer, { count: 0 });
dispatch({ type: 'increment' });
üí° Use useReducer for complex state logic or when next state depends on previous state!
üìù useRef Hook
const ref = useRef(initialValue)
Create mutable ref object that persists across renders
// Access DOM element
const inputRef = useRef(null);
inputRef.current.focus();
// Store mutable value (doesn't trigger re-render)
const countRef = useRef(0);
countRef.current += 1;
// Store previous value
const prevValue = useRef();
useEffect(() => {
prevValue.current = value;
}, [value]);
üé® useMemo Hook
const memoizedValue = useMemo(() => compute(), [deps])
Memoize expensive computations
// Expensive calculation
const expensiveValue = useMemo(() => {
return computeExpensiveValue(a, b);
}, [a, b]);
// Memoize filtered list
const filteredList = useMemo(() => {
return list.filter(item => item.includes(query));
}, [list, query]);
Note: Only use useMemo for expensive computations. Premature optimization can hurt performance!
‚öôÔ∏è useCallback Hook
const memoizedFn = useCallback(() => {}, [deps])
Memoize callback functions to prevent unnecessary re-renders
// Memoize callback
const handleClick = useCallback(() => {
console.log('Clicked with count:', count);
}, [count]);
// Pass to child component
<ChildComponent onClick={handleClick} />
üí° Use useCallback when passing callbacks to optimized child components wrapped in React.memo!
üîß useLayoutEffect Hook
useLayoutEffect(() => { /* effect */ }, [deps])
Run effects synchronously after DOM mutations
// Measure DOM before paint
useLayoutEffect(() => {
const rect = divRef.current.getBoundingClientRect();
setHeight(rect.height);
}, []);
‚ö†Ô∏è useLayoutEffect blocks visual updates. Use useEffect unless you need to measure/mutate DOM before paint!

üß© Component Composition
// Render props pattern
function DataProvider({ render }) {
const [data, setData] = useState(null);
return render(data);
}
<DataProvider render={(data) => <Display data={data} />} />
// Higher-Order Component (HOC)
function withLoading(Component) {
return function WithLoadingComponent({ isLoading, ...props }) {
if (isLoading) return <Loading />;
return <Component {...props} />;
};
}
// Custom Hook pattern
function useLocalStorage(key, initialValue) {
const [value, setValue] = useState(() => {
const item = localStorage.getItem(key);
return item ? JSON.parse(item) : initialValue;
});
useEffect(() => {
localStorage.setItem(key, JSON.stringify(value));
}, [key, value]);
return [value, setValue];
}

üõ°Ô∏è Error Boundaries
// Error boundary class component
class ErrorBoundary extends React.Component {
constructor(props) {
super(props);
this.state = { hasError: false };
}
static getDerivedStateFromError(error) {
return { hasError: true };
}
componentDidCatch(error, errorInfo) {
console.error('Error:', error, errorInfo);
}
render() {
if (this.state.hasError) {
return <h1>Something went wrong.</h1>;
}
return this.props.children;
}
}
// Usage
<ErrorBoundary>
<MyComponent />
</ErrorBoundary>
Note: Error boundaries only catch errors in child components, not in event handlers or async code.
üé¨ Component Lifecycle (Class Components)
componentDidMount()
Called after component is mounted (similar to useEffect with [])
componentDidUpdate(prevProps, prevState)
Called after component updates (similar to useEffect with deps)
componentWillUnmount()
Called before component is unmounted (similar to useEffect cleanup)
shouldComponentUpdate(nextProps, nextState)
Return false to prevent re-render (similar to React.memo)
getDerivedStateFromProps(props, state)
Update state based on props changes (rarely needed)
üí° Functional components with hooks are now preferred over class components!

üí° Best Practices
üí° Keep Components Small: Break down large components into smaller, reusable pieces for better maintainability.
üí° Use Functional Components: Prefer functional components with hooks over class components for cleaner code.
üí° Avoid Inline Functions in JSX: Define event handlers outside JSX or use useCallback to prevent unnecessary re-renders.
üí° Destructure Props: Destructure props in function parameters for cleaner code and better readability.
üí° Use Key Prop Correctly: Always provide stable, unique keys when rendering lists to help React optimize updates.
‚ö†Ô∏è Don't Mutate State: Always create new objects/arrays when updating state. Never modify state directly!
‚ö†Ô∏è Avoid Index as Key: Don't use array index as key if the list can be reordered or items can be added/removed.
‚ö†Ô∏è Clean Up Effects: Always return cleanup functions from useEffect to prevent memory leaks.
üöÄ Common Patterns
// Pattern 1: Fetch data on mount
useEffect(() => {
async function fetchData() {
const response = await fetch('/api/data');
const data = await response.json();
setData(data);
}
fetchData();
}, []);
// Pattern 2: Debounced input
const [searchTerm, setSearchTerm] = useState('');
const [debouncedTerm, setDebouncedTerm] = useState(searchTerm);
useEffect(() => {
const timer = setTimeout(() => {
setDebouncedTerm(searchTerm);
}, 500);
return () => clearTimeout(timer);
}, [searchTerm]);
// Pattern 3: Toggle state
const [isOpen, setIsOpen] = useState(false);
const toggle = () => setIsOpen(prev => !prev);
// Pattern 4: Previous value
function usePrevious(value) {
const ref = useRef();
useEffect(() => {
ref.current = value;
}, [value]);
return ref.current;
}
// Pattern 5: Window size hook
function useWindowSize() {
const [size, setSize] = useState({
width: window.innerWidth,
height: window.innerHeight
});
useEffect(() => {
const handleResize = () => {
setSize({ width: window.innerWidth, height: window.innerHeight });
};
window.addEventListener('resize', handleResize);
return () => window.removeEventListener('resize', handleResize);
}, []);
return size;
}
// Pattern 6: Form with multiple inputs
const [formData, setFormData] = useState({
name: '',
email: '',
age: 0
});
const handleChange = (e) => {
setFormData({
...formData,
[e.target.name]: e.target.value
});
};
