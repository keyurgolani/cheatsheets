üí° CSS Variables (Custom Properties)

--primary-color: #667eea;
--spacing: 1rem;
--border-radius: 8px;
}
/* Use variables */

background: var(--primary-color);
padding: var(--spacing);
border-radius: var(--border-radius);
}
/* Fallback values */
color: var(--text-color, #333);
