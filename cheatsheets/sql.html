üìä SELECT Queries
-- Basic select
SELECT * FROM users;
-- Select specific columns
SELECT name, email FROM users;
-- With alias
SELECT name AS full_name FROM users;
-- Distinct values
SELECT DISTINCT city FROM users;
-- Limit results
SELECT * FROM users LIMIT 10;
üîç WHERE Clause
-- Basic conditions
SELECT * FROM users
WHERE age > 18;
-- Multiple conditions
SELECT * FROM users
WHERE age > 18 AND city = 'NYC';
-- OR condition
WHERE city = 'NYC' OR city = 'LA';
-- IN operator
WHERE city IN ('NYC', 'LA', 'SF');
-- BETWEEN
WHERE age BETWEEN 18 AND 65;
-- LIKE pattern matching
WHERE name LIKE 'J%';
WHERE email LIKE '%@gmail.com';
-- IS NULL
WHERE phone IS NULL;
WHERE phone IS NOT NULL;
üìà ORDER BY & Sorting
-- Sort ascending (default)
SELECT * FROM users
ORDER BY name;
-- Sort descending
ORDER BY age DESC;
-- Multiple columns
ORDER BY city ASC, age DESC;
üìä Aggregate Functions
-- COUNT
SELECT COUNT(*) FROM users;
SELECT COUNT(DISTINCT city) FROM users;
-- SUM, AVG, MIN, MAX
SELECT SUM(price) FROM orders;
SELECT AVG(age) FROM users;
SELECT MIN(price), MAX(price)
FROM products;
üîó GROUP BY & HAVING
-- Group by single column
SELECT city, COUNT(*)
FROM users
GROUP BY city;
-- Multiple columns
SELECT city, age, COUNT(*)
FROM users
GROUP BY city, age;
-- HAVING (filter groups)
SELECT city, COUNT(*) AS cnt
FROM users
GROUP BY city
HAVING cnt > 10;
üîÑ JOIN Operations
-- INNER JOIN
SELECT users.name, orders.amount
FROM users
INNER JOIN orders
ON users.id = orders.user_id;
-- LEFT JOIN
SELECT users.name, orders.amount
FROM users
LEFT JOIN orders
ON users.id = orders.user_id;
-- RIGHT JOIN
RIGHT JOIN orders
ON users.id = orders.user_id;
-- FULL OUTER JOIN
FULL OUTER JOIN orders
ON users.id = orders.user_id;

üóëÔ∏è DELETE Data
-- Delete specific rows

üèóÔ∏è CREATE TABLE
CREATE TABLE users (
id INT PRIMARY KEY AUTO_INCREMENT,
name VARCHAR(100) NOT NULL,
email VARCHAR(255) UNIQUE,
age INT,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
-- With foreign key
CREATE TABLE orders (
id INT PRIMARY KEY,
user_id INT,
amount DECIMAL(10, 2),
FOREIGN KEY (user_id)
REFERENCES users(id)
);

üìë Indexes
-- Create index
CREATE INDEX idx_email
ON users(email);
-- Unique index
CREATE UNIQUE INDEX idx_email
ON users(email);
-- Composite index
CREATE INDEX idx_name_city
ON users(name, city);
-- Drop index
DROP INDEX idx_email ON users;
üîç Subqueries
-- Subquery in WHERE
SELECT * FROM users
WHERE age > (SELECT AVG(age) FROM users);
-- Subquery in FROM
SELECT avg_age
FROM (SELECT AVG(age) AS avg_age FROM users) AS subquery;
-- Correlated subquery
SELECT name
FROM users u
WHERE age > (SELECT AVG(age)
FROM users u2
WHERE u2.city = u.city);
