// Fixed window size
function maxSumSubarray(arr, k):
windowSum = sum(arr[0:k])
maxSum = windowSum
for i = k to length(arr) - 1:
windowSum = windowSum - arr[i - k] + arr[i]
maxSum = max(maxSum, windowSum)
return maxSum
// Variable window size
function longestSubstringKDistinct(s, k):
left = 0, maxLen = 0
charCount = {}
for right = 0 to length(s) - 1:
charCount[s[right]]++
while size(charCount) > k:
charCount[s[left]]--
if charCount[s[left]] == 0:
remove s[left] from charCount
left++
maxLen = max(maxLen, right - left + 1)
return maxLen

// Opposite direction (converging)
function twoSum(arr, target):
left = 0
right = length(arr) - 1
while left < right:
sum = arr[left] + arr[right]
if sum == target:
return [left, right]
else if sum < target:
left++
else:
right--
return null
// Same direction (chasing)
function removeDuplicates(arr):
if length(arr) == 0:
return 0
slow = 0
for fast = 1 to length(arr) - 1:
if arr[fast] != arr[slow]:
slow++
arr[slow] = arr[fast]
return slow + 1

// Cycle detection
function hasCycle(head):
slow = head
fast = head
while fast != null and fast.next != null:
slow = slow.next
fast = fast.next.next
if slow == fast:
return true
return false
// Find middle of linked list
function findMiddle(head):
slow = head
fast = head
while fast != null and fast.next != null:
slow = slow.next
fast = fast.next.next
return slow // Middle node
// Find cycle start
function findCycleStart(head):
slow = fast = head
// Find meeting point
while fast and fast.next:
slow = slow.next
fast = fast.next.next
if slow == fast:
break
// Find cycle start
slow = head
while slow != fast:
slow = slow.next
fast = fast.next
return slow
üìÖ Merge Intervals
// Merge overlapping intervals
function mergeIntervals(intervals):
if length(intervals) <= 1:
return intervals
// Sort by start time
sort(intervals, key=start)
merged = [intervals[0]]
for i = 1 to length(intervals) - 1:
current = intervals[i]
last = merged[length(merged) - 1]
if current.start <= last.end:
// Overlapping, merge them
last.end = max(last.end, current.end)
else:
// Non-overlapping, add to result
merged.append(current)
return merged
// Insert interval
function insertInterval(intervals, newInterval):
result = []
i = 0
// Add all intervals before newInterval
while i < length(intervals) and intervals[i].end < newInterval.start:
result.append(intervals[i])
i++
// Merge overlapping intervals
while i < length(intervals) and intervals[i].start <= newInterval.end:
newInterval.start = min(newInterval.start, intervals[i].start)
newInterval.end = max(newInterval.end, intervals[i].end)
i++
result.append(newInterval)
// Add remaining intervals
while i < length(intervals):
result.append(intervals[i])
i++
return result
üîÑ Cyclic Sort
// Cyclic sort - place each number at its correct index
function cyclicSort(arr):
i = 0
while i < length(arr):
correctIndex = arr[i] - 1
if arr[i] != arr[correctIndex]:
swap(arr[i], arr[correctIndex])
else:
i++
return arr
// Find missing number
function findMissingNumber(arr):
i = 0
n = length(arr)
// Place each number at index equal to its value
while i < n:
correctIndex = arr[i]
if arr[i] < n and arr[i] != arr[correctIndex]:
swap(arr[i], arr[correctIndex])
else:
i++
// Find the first index where number doesn't match
for i = 0 to n - 1:
if arr[i] != i:
return i
return n
// Find all duplicates
function findDuplicates(arr):
i = 0
while i < length(arr):
correctIndex = arr[i] - 1
if arr[i] != arr[correctIndex]:
swap(arr[i], arr[correctIndex])
else:
i++
duplicates = []
for i = 0 to length(arr) - 1:
if arr[i] != i + 1:
duplicates.append(arr[i])
return duplicates

// Reverse entire linked list
function reverseList(head):
prev = null
current = head
while current != null:
next = current.next
current.next = prev
prev = current
current = next
return prev // New head
// Reverse sublist from position m to n
function reverseBetween(head, m, n):
if m == n:
return head
// Skip to position m-1
current = head
prev = null
for i = 0 to m - 1:
prev = current
current = current.next
// Reverse from m to n
lastNodeBeforeReverse = prev
lastNodeOfReverse = current
for i = 0 to n - m:
next = current.next
current.next = prev
prev = current
current = next
// Connect with remaining list
if lastNodeBeforeReverse != null:
lastNodeBeforeReverse.next = prev
else:
head = prev
lastNodeOfReverse.next = current
return head
// Reverse every k nodes
function reverseKGroup(head, k):
if k <= 1 or head == null:
return head
current = head
prev = null
while true:
lastNodeBeforeReverse = prev
lastNodeOfReverse = current
// Reverse k nodes
for i = 0 to k - 1 and current != null:
next = current.next
current.next = prev
prev = current
current = next
// Connect with previous part
if lastNodeBeforeReverse != null:
lastNodeBeforeReverse.next = prev
else:
head = prev
lastNodeOfReverse.next = current
if current == null:
break
prev = lastNodeOfReverse
return head
