#!/bin/bash Scripting Essentials

Complete guide to shell scripting in Bash

üöÄ Script Template
#!/bin/bash
# Script: script_name.sh
# Description: What this script does
# Author: Your Name
# Date: 2025-01-01
# Exit on error, undefined variables, pipe failures
set -euo pipefail
# Main script logic here
echo "Hello, World!"
üìù Variables
# Variable assignment (no spaces!)
NAME="John"
AGE=30
# Using variables
echo "$NAME"
echo "${NAME}" # preferred
# Command substitution
CURRENT_DATE=$(date)
FILES=`ls` # old style
# Read-only variable
readonly PI=3.14159
# Environment variable
export PATH=$PATH:/new/path
üî¢ Special Variables
$0
Script name
$1, $2, $3...
Positional parameters (arguments)
$#
Number of arguments
$@
All arguments (as separate words)
$*
All arguments (as single word)
$?
Exit status of last command
$$
Process ID of current shell
$!
Process ID of last background command
üéØ String Operations
STRING="Hello World"
# Length
echo ${#STRING} # 11
# Substring
echo ${STRING:0:5} # Hello
# Replace first occurrence
echo ${STRING/World/Universe}
# Replace all occurrences
echo ${STRING//o/0}
# Default value if empty
echo ${VAR:-"default"}
# Convert to uppercase/lowercase
echo ${STRING^^} # HELLO WORLD
echo ${STRING,,} # hello world
üîç Conditional Statements
# if-else
if [ $AGE -gt 18 ]; then
echo "Adult"
elif [ $AGE -eq 18 ]; then
echo "Just turned adult"
else
echo "Minor"
fi
# One-liner
[ $AGE -gt 18 ] && echo "Adult" || echo "Minor"
# Modern test syntax
if [[ $NAME == "John" ]]; then
echo "Found John"
fi
‚öñÔ∏è Test Operators
-eq, -ne, -lt, -le, -gt, -ge
Numeric comparisons (equal, not equal, less than, etc.)
=, !=, <, >
String comparisons
-z, -n
String is empty, string is not empty
-e, -f, -d
File exists, is regular file, is directory
-r, -w, -x
File is readable, writable, executable
-s
File exists and is not empty
&&, ||, !
Logical AND, OR, NOT
üîÑ Loops
# For loop
for i in 1 2 3 4 5; do
echo "Number: $i"
done
# For loop with range
for i in {1..10}; do
echo $i
done
# C-style for loop
for ((i=0; i<10; i++)); do
echo $i
done
# While loop
COUNT=0
while [ $COUNT -lt 5 ]; do
echo $COUNT
((COUNT++))
done
# Until loop
until [ $COUNT -eq 10 ]; do
echo $COUNT
((COUNT++))
done
üìã Arrays
# Define array
FRUITS=("Apple" "Banana" "Orange")
# Access elements
echo ${FRUITS[0]} # Apple
echo ${FRUITS[@]} # All elements
echo ${#FRUITS[@]} # Array length
# Add element
FRUITS+=("Grape")
# Loop through array
for fruit in "${FRUITS[@]}"; do
echo $fruit
done
# Associative array (hash/dict)
declare -A COLORS
COLORS[red]="#FF0000"
COLORS[green]="#00FF00"
echo ${COLORS[red]}
üîß Functions
# Define function
function greet() {
echo "Hello, $1!"
}
# Alternative syntax
greet() {
local NAME=$1 # local variable
echo "Hello, $NAME!"
}
# Call function
greet "World"
# Return value
add() {
local result=$(($1 + $2))
echo $result
}
SUM=$(add 5 3)
echo $SUM # 8
üì• User Input
# Read input
echo "Enter your name:"
read NAME
echo "Hello, $NAME!"
# Read with prompt
read -p "Enter age: " AGE
# Read password (hidden)
read -sp "Enter password: " PASSWORD
echo
# Read with timeout
read -t 5 -p "Quick! Enter something: " INPUT
# Read file line by line
while IFS= read -r line; do
echo "Line: $line"
done < file.txt
üîÄ Case Statement
case $1 in
start)
echo "Starting..."
;;
stop)
echo "Stopping..."
;;
restart)
echo "Restarting..."
;;
*)
echo "Usage: $0 {start|stop|restart}"
exit 1
;;
esac
üì§ Input/Output Redirection
# Redirect output
echo "Hello" > file.txt
# Append output
echo "World" >> file.txt
# Redirect stderr
command 2> errors.txt
# Redirect both stdout and stderr
command &> output.txt
command > output.txt 2>&1
# Discard output
command > /dev/null 2>&1
# Here document
cat << EOF
Multiple lines
of text
EOF
üî¢ Arithmetic
# Using (( ))
((x = 5 + 3))
((x++))
((x += 5))
# Using expr
x=$(expr 5 + 3)
# Using $[ ]
x=$[5 + 3]
# Using $(( ))
x=$((5 + 3))
y=$((x * 2))
# Floating point with bc
result=$(echo "scale=2; 10 / 3" | bc)
üí° Practical Examples
# Check if command exists
if command -v git &> /dev/null; then
echo "Git is installed"
fi
# Parse command line options
while getopts "hv:f:" opt; do
case $opt in
h) echo "Help message" ;;
v) VERSION=$OPTARG ;;
f) FILE=$OPTARG ;;
*) exit 1 ;;
esac
done
# Progress bar
for i in {1..100}; do
printf "\rProgress: [%-50s] %d%%" $(printf '#%.0s' $(seq 1 $((i/2)))) $i
sleep 0.05
done
echo
# Error handling with trap
cleanup() {
echo "Cleaning up..."
rm -f /tmp/tempfile
}
trap cleanup EXIT
# Check if script is run as root
if [ "$(id -u)" -ne 0 ]; then
echo "This script must be run as root"
exit 1
fi

‚Ä¢ Use functions for repeated code
‚Ä¢ Add comments and documentation
