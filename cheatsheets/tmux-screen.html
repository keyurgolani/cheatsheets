# ~/.tmux.conf
# Change prefix to Ctrl+a
unbind C-b
set -g prefix C-a
bind C-a send-prefix
# Enable mouse support
set -g mouse on
# Start windows at 1
set -g base-index 1
# Split panes with | and -
bind | split-window -h
bind - split-window -v
# Reload config
bind r source-file ~/.tmux.conf
