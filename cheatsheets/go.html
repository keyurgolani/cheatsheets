}
return total
}

Message string
}

}
// Check error type

}
üöÄ Goroutines

}

üì° Channels
// Create channel
ch := make(chan int)
// Send to channel
ch <- 42
// Receive from channel
value := <-ch
// Buffered channel
ch := make(chan int, 10) // buffer size 10
// Close channel
close(ch)
// Check if channel is closed
value, ok := <-ch
if !ok {
fmt.Println("Channel closed")
}
// Range over channel
for value := range ch {
fmt.Println(value)
}
// Directional channels
func send(ch chan<- int) { // send-only
ch <- 42
}
func receive(ch <-chan int) { // receive-only
value := <-ch
}
üîÄ Select Statement
// Select waits on multiple channels
select {
case msg1 := <-ch1:
fmt.Println("Received from ch1:", msg1)
case msg2 := <-ch2:
fmt.Println("Received from ch2:", msg2)
case ch3 <- 42:
fmt.Println("Sent to ch3")

fmt.Println("No channel ready")
}
// Timeout pattern
select {
case result := <-ch:
fmt.Println(result)
case <-time.After(5 * time.Second):
fmt.Println("Timeout!")
}
// Non-blocking receive
select {
case msg := <-ch:
fmt.Println(msg)
default:
fmt.Println("No message")
}
üîí Mutex & Synchronization
import "sync"
// Mutex for safe concurrent access
type SafeCounter struct {
mu sync.Mutex
count int
}
func (c *SafeCounter) Inc() {
c.mu.Lock()
defer c.mu.Unlock()
c.count++
}
func (c *SafeCounter) Value() int {
c.mu.Lock()
defer c.mu.Unlock()
return c.count
}
// RWMutex for read-heavy workloads
var rwMutex sync.RWMutex
// Multiple readers allowed
rwMutex.RLock()
defer rwMutex.RUnlock()
// Exclusive writer
rwMutex.Lock()
defer rwMutex.Unlock()
üéØ Concurrency Patterns
// Worker Pool Pattern
func worker(id int, jobs <-chan int, results chan<- int) {
for job := range jobs {
results <- job * 2
}
}
jobs := make(chan int, 100)
results := make(chan int, 100)
// Start workers
for w := 1; w <= 3; w++ {
go worker(w, jobs, results)
}
// Pipeline Pattern
func gen(nums ...int) <-chan int {
out := make(chan int)
go func() {
for _, n := range nums {
out <- n
}
close(out)
}()
return out
}
func square(in <-chan int) <-chan int {
out := make(chan int)
go func() {
for n := range in {
out <- n * n
}
close(out)
}()
return out
}

‚è∞ Context Package
import "context"
// Background context
ctx := context.Background()
// With timeout
ctx, cancel := context.WithTimeout(
context.Background(),
5*time.Second,
)
defer cancel()
// With cancellation
ctx, cancel := context.WithCancel(context.Background())
go func() {
<-time.After(2 * time.Second)
cancel() // Cancel after 2 seconds
}()
// With value
ctx = context.WithValue(ctx, "userID", 123)
userID := ctx.Value("userID").(int)
// Check cancellation
select {
case <-ctx.Done():
fmt.Println("Context cancelled")
return
default:
// Continue work
}

// Table-driven tests
func TestAddTable(t *testing.T) {
tests := []struct {
a, b, want int
}{
{1, 2, 3},
{0, 0, 0},
{-1, 1, 0},
}
for _, tt := range tests {
got := Add(tt.a, tt.b)
if got != tt.want {
t.Errorf("Add(%d, %d) = %d; want %d",
tt.a, tt.b, got, tt.want)
}
}
}
// Benchmark
func BenchmarkAdd(b *testing.B) {
for i := 0; i < b.N; i++ {
Add(2, 3)
}
}
