üìã Interfaces
// Define interface
type Shape interface {

Perimeter() float64
}
// Implement interface (implicit)
type Rectangle struct {
Width, Height float64
}

}

}

case string:
fmt.Println("string:", v)
}

‚ö†Ô∏è Error Handling
// Functions return errors
func readFile(path string) ([]byte, error) {
data, err := os.ReadFile(path)

return nil, err
}

}
// Create custom error
import "errors"
err := errors.New("something went wrong")
// Formatted error
err := fmt.Errorf("failed to open %s: %w", path, err)
// Custom error type
type MyError struct {
Code int
Message string
}
func (e *MyError) Error() string {
return fmt.Sprintf("error %d: %s", e.Code, e.Message)
}
// Check error type
if errors.Is(err, os.ErrNotExist) {
fmt.Println("file not found")
}
‚ö†Ô∏è Always check errors! Go doesn't have exceptions - errors are values

üì° Channels
// Create channel
ch := make(chan int)
// Send to channel
ch <- 42
// Receive from channel
value := <-ch
// Buffered channel
ch := make(chan int, 10) // buffer size 10
// Close channel
close(ch)
// Check if channel is closed
value, ok := <-ch
if !ok {
fmt.Println("Channel closed")
}
// Range over channel
for value := range ch {
fmt.Println(value)
}
// Directional channels
func send(ch chan<- int) { // send-only
ch <- 42
}
func receive(ch <-chan int) { // receive-only
value := <-ch
}
üîÄ Select Statement
// Select waits on multiple channels
select {
case msg1 := <-ch1:
fmt.Println("Received from ch1:", msg1)
case msg2 := <-ch2:
fmt.Println("Received from ch2:", msg2)
case ch3 <- 42:
fmt.Println("Sent to ch3")
default:
fmt.Println("No channel ready")
}
// Timeout pattern
select {
case result := <-ch:
fmt.Println(result)
case <-time.After(5 * time.Second):
fmt.Println("Timeout!")
}
// Non-blocking receive
select {
case msg := <-ch:
fmt.Println(msg)
default:
fmt.Println("No message")
}
Select behavior: If multiple cases are ready, one is chosen randomly
üîí Mutex & Synchronization
import "sync"
// Mutex for safe concurrent access
type SafeCounter struct {
mu sync.Mutex
count int
}
func (c *SafeCounter) Inc() {
c.mu.Lock()
defer c.mu.Unlock()
c.count++
}
func (c *SafeCounter) Value() int {
c.mu.Lock()
defer c.mu.Unlock()
return c.count
}
// RWMutex for read-heavy workloads
var rwMutex sync.RWMutex
// Multiple readers allowed
rwMutex.RLock()
defer rwMutex.RUnlock()
// Exclusive writer
rwMutex.Lock()
defer rwMutex.Unlock()
üí° Use defer to ensure mutex is always unlocked
üéØ Concurrency Patterns
// Worker Pool Pattern
func worker(id int, jobs <-chan int, results chan<- int) {
for job := range jobs {
results <- job * 2
}
}
jobs := make(chan int, 100)
results := make(chan int, 100)
// Start workers
for w := 1; w <= 3; w++ {
go worker(w, jobs, results)
}
// Pipeline Pattern
func gen(nums ...int) <-chan int {
out := make(chan int)
go func() {
for _, n := range nums {
out <- n
}
close(out)
}()
return out
}
func square(in <-chan int) <-chan int {
out := make(chan int)
go func() {
for n := range in {
out <- n * n
}
close(out)
}()
return out
}

üí° Best Practices
Naming Conventions:
‚Ä¢ Exported names start with uppercase (e.g., PublicFunc)
‚Ä¢ Unexported names start with lowercase (e.g., privateFunc)
‚Ä¢ Use camelCase for multi-word names
‚Ä¢ Keep names short and meaningful
‚Ä¢ Interface names often end in -er (e.g., Reader, Writer)
Concurrency:
‚Ä¢ Don't communicate by sharing memory; share memory by communicating
‚Ä¢ Use channels to pass ownership of data
‚Ä¢ Use mutexes to protect shared state
‚Ä¢ Always close channels when done sending
‚Ä¢ Use select for multiplexing channels
‚Ä¢ Use context for cancellation and timeouts
Error Handling:
‚Ä¢ Check errors immediately after they occur
‚Ä¢ Don't ignore errors (use _ only when intentional)
‚Ä¢ Wrap errors with context using fmt.Errorf with %w
‚Ä¢ Return errors as the last return value
‚Ä¢ Use custom error types for complex error handling
Code Organization:
‚Ä¢ Keep packages focused and cohesive
‚Ä¢ Use internal/ for private packages
‚Ä¢ Run go fmt before committing
‚Ä¢ Use go vet to catch common mistakes
‚Ä¢ Write tests alongside your code
‚Ä¢ Document exported functions and types
