üìÑ Dockerfile Instructions
# Base image
FROM ubuntu:20.04
# Set metadata
LABEL maintainer="user@example.com"
# Set environment variables
ENV NODE_ENV=production
ENV PORT=3000
# Set working directory
WORKDIR /app
# Copy files
COPY package*.json ./
COPY . .
# Add files (with extraction for tar)
ADD archive.tar.gz /app
# Run commands
RUN apt-get update && apt-get install -y nodejs
RUN npm install
# Expose port
EXPOSE 3000
# Create volume mount point
VOLUME ["/data"]
# Set user
USER node
# Default command (can be overridden)
CMD ["npm", "start"]
# Entrypoint (not easily overridden)
ENTRYPOINT ["node", "server.js"]
üé® Dockerfile Best Practices
# Use multi-stage builds
FROM node:16 AS builder
WORKDIR /app
COPY package*.json ./
RUN npm install
COPY . .
RUN npm run build
FROM node:16-alpine
WORKDIR /app
COPY --from=builder /app/dist ./dist
COPY package*.json ./
RUN npm install --production
CMD ["node", "dist/index.js"]

‚Ä¢ Use specific image tags, not latest

‚Ä¢ Order instructions from least to most frequently changing
‚Ä¢ Use .dockerignore to exclude files
‚Ä¢ Leverage build cache for faster builds
